package musicmj;

import javafx.scene.Scene;
import javafx.scene.image.Image;
import javafx.scene.image.ImageView;
import javafx.stage.Stage;
import javafx.scene.layout.ClipView;
import javafx.animation.KeyFrame;
import javafx.animation.Timeline;

import javafx.scene.paint.Color;

var imageView = ImageView {
   image: Image {
       url: "{__DIR__}image.jpg"
   }
}

var clipView = ClipView {
    clipX: 0
    clipY: 0
    width: 166
    height: 200
    node: imageView
    pannable: false
    translateX: 10
    translateY: 10
}

var timeline = Timeline {
    
    repeatCount: Timeline.INDEFINITE

    keyFrames: KeyFrame {

        time: 500ms

        action: function() {

            if((clipView.clipY >= 400) and (clipView.clipX >= 800)) {
                clipView.clipX = 0;
                clipView.clipY = 0;
            } else if(clipView.clipX >= 800) {
                clipView.clipX = 0;
                clipView.clipY += clipView.height + 9;
            }

            clipView.clipX += clipView.width + 7;
        }
    }
}
timeline.play();

Stage {

    title: "King of Pop"
    
    scene: Scene {
        content: [
            clipView
        ]
        width: 186
        height: 220
        fill: Color.WHITE
    }

    resizable: false
}
