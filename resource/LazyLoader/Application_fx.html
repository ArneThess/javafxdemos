package lazyloader;

import javafx.async.JavaTaskBase;
import javafx.async.RunnableFuture;
import javax.swing.JComponent;

/**
 * @author Rakesh Menon
 */

public class Application extends JavaTaskBase, CallbackHandler {
    
    public-init var className : String;
    public-init var urls : String[];
    public-init var onLoad: function(jComponent : JComponent): Void = null;
    
    override function create() : RunnableFuture {
        return new LazyClassLoader(this);
    }

    override function getClassName() : String {
        return className;
    }

    override function getURLList() : java.util.List {
        
        var list = new java.util.ArrayList();
        for(url in urls) {
            list.add(url);
        }
        return list;
    }
    
    override function onLaunch(jComponent : JComponent) : Void {
        if(onLoad == null) { return; }
        onLoad(jComponent);
    }
}
