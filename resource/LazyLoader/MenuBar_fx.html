package lazyloader.view;

import javafx.scene.CustomNode;
import javafx.scene.Group;
import javafx.scene.Node;
import javafx.scene.paint.Color;
import javafx.scene.paint.LinearGradient;
import javafx.scene.shape.Rectangle;
import javafx.scene.paint.Stop;
import javafx.scene.layout.VBox;
import javafx.geometry.Bounds;
import javafx.animation.transition.TranslateTransition;
import javafx.scene.effect.DropShadow;
 
public class MenuBar extends CustomNode {

    public-init var content : Menu[];

    def spacing = 5.0;

    var selRectangle = Rectangle {
        x: -2
        y: -2
        width: Menu.WIDTH + 4
        height: Menu.HEIGHT + 4
        fill: LinearGradient {
            startX: 0.0, startY: 0.0, endX: 0.0, endY: 1.0
            proportional: true
            stops: [
                Stop { offset: 0.0 color: Color.ALICEBLUE },
                Stop { offset: 0.5 color: Color.WHITE },
                Stop { offset: 1.0 color: Color.SILVER }
            ]
        }
        opacity: 0.5
        stroke: LinearGradient {
            startX: 0.0, startY: 0.0, endX: 0.0, endY: 1.0
            proportional: true
            stops: [
                Stop { offset: 0.0 color: Color.SILVER },
                Stop { offset: 0.5 color: Color.ALICEBLUE },
                Stop { offset: 1.0 color: Color.SILVER }
            ]
        }
        strokeWidth: 2.0
        arcWidth: 10
        arcHeight: 10
        effect: DropShadow { }
    };

    public-read var adjustSelection = TranslateTransition {
        duration: 500ms
        node: selRectangle
    }
    
    override function create() : Node {

        // Update MenuBar
        for(menu in content) {
            menu.menuBar = this;
        }

        var vBoxText = VBox {
            content: content
        };
        
        var menuGroup = Group {
            translateX: 10
            translateY: 10
            content: [ selRectangle, vBoxText ]
        }
        
        return Group {
            content: [ menuGroup ]
        }
    }

    public function updateSelRect(boundsInScene : Bounds) {
        var localBounds = selRectangle.sceneToLocal(boundsInScene);
        adjustSelection.byY = localBounds.minY;
        adjustSelection.playFromStart();
    }
}
