var svgPath: SVGPath[];
var selSVGPath: SVGPath;

// Handle SVGPath selected event
function onSVGPathSelect( e: MouseEvent) : Void {
    
    // Reset stroke color
    if(selSVGPath != null) {
        selSVGPath.stroke = null;
    }
    
    selSVGPath = e.node as SVGPath;
    selSVGPath.stroke = Color.BLUE;
}

// Handle ColorChooser selection
function onColorChooserSelect( e: MouseEvent) : Void {

    // No node selected
    if(selSVGPath == null) {
        return;
    }
    
    if(selSVGPath.fill instanceof Color) {
        selSVGPath.fill = colorChooser.color;
    } else {
        selSVGPath.fill = RadialGradient {
            centerX: 0.75
            centerY: 0.5
            radius: 1.0
            stops: [
                Stop {
                    offset: 0.0
                    color: Color.LIGHTGRAY
                },
                Stop {
                    offset: 0.5
                    color: colorChooser.color;
                }
            ]
        }
    }
}
