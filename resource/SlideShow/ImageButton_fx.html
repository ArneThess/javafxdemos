package slideshow;

import javafx.scene.image.Image;
import javafx.scene.image.ImageView;

// Custom button created using Images
public class ImageButton extends ImageView { 
    
    public var normalImage: Image on replace {
        if(normalImage != null) { 
            image = normalImage;
        }
    }
    public var hoverImage: Image;
        
    public override var onMouseEntered = function(e) {
        if(hoverImage != null) { image = hoverImage; }
    }
    
    public override var onMouseExited = function(e) {
        if(normalImage != null) { image = normalImage; }
    }
}
